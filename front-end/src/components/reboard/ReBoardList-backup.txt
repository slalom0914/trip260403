import React, { useEffect, useState } from 'react'
import Header from '../include/Header'
import Footer from '../include/Footer'
import styles from './reboard.module.css'
import { useNavigate } from 'react-router-dom'
import { boardListDB } from '../../service/boardLogic'
import ReBoardItem from './ReBoardItem'
const ReBoardList = () => {
  const navigate = useNavigate()
  
  const [boards, setBoards] = useState([])
  const boardSearch = async () => {
    const gubun = document.querySelector("#gubun").value
    const keyword = document.querySelector("#keyword").value
    console.log(`${gubun}, ${keyword}`)
    const board = { gubun, keyword }
    try {
      const res = await boardListDB(board)
      console.log(res.data)
      setBoards(res.data)
    } catch (e) {
      console.error(e)
      const msg = e.response?.status === 500
        ? '서버 오류(500): 백엔드 로그의 SQL·DB 연결을 확인하세요.'
        : (e.message || '목록 요청 실패')
      alert(msg)
    }
  }
  const boardList = async () => {
    console.log("전체조회")
    const board = { gubun: null, keyword: null }
    try {
      const res = await boardListDB(board)
      setBoards(res.data)
    } catch (e) {
      console.error(e)
      alert(e.response?.status === 500 ? '서버 오류(500)' : (e.message || '목록 요청 실패'))
    }
  }
  //목록 페이지를 열자 마자 DB경유하는 코드는 어디서 어떻게 작성할까?
  useEffect (() => {
    const asyncDB = async() => {
        const board = { gubun: null, keyword: null }
        try {
          const res = await boardListDB(board)
          console.log(res.data)
          setBoards(res.data)
        } catch (e) {
          console.error(e)
        }
    }
    asyncDB()
  },[]) //의존성 배열에 useState넣을 때는 무한루프에 빠질 수 있다  
  return (
    <>
      <Header />
      <div className="container">
        <div className={styles.pageHeader}>
          <h2>댓글게시판<small> - Quill Editor</small></h2>
          <hr />
        </div>
        <form
          className="row"
          onSubmit={(e) => {
            e.preventDefault()
            boardSearch()
          }}
        >
          <div className="col-sm-3">
            <select className="form-select" id="gubun">
              <option value="">분류선택</option>
              <option value="b_title">제목</option>
              <option value="b_writer">작성자</option>
              <option value="b_content">내용</option>
            </select>
          </div>
          <div className="col-sm-6">
            <input
              type="text"
              className="form-control"
              placeholder="검색어를 입력하세요"
              id="keyword"
              autoComplete="off"
            />
          </div>
          <div className="col-sm-3">
            <button type="submit" className="btn btn-danger">검색</button>
          </div>
        </form>
        <div className="row">
          <div className="col-sm-12">
            <table className={`table table-hover ${styles.listBody}`}>
              <thead>
                <tr>
                  <th>#</th>
                  <th>제목</th>
                  <th>작성자</th>
                  <th>작성일</th>
                </tr>
              </thead>
              {/* 데이터셋 연동하기 */}
              {/* props로 넘어온 상태값이 빈 깡통이면 실행하지 않기 */}
              <tbody>
                {Array.isArray(boards) && boards.map((board, index) => (
                  <ReBoardItem key={board.b_no} {...board} />
                ))}
              </tbody>
              {/* 데이터셋 연동하기 */}
            </table>        
            <div className={styles.listFooter}>
              <button className="btn btn-warning" onClick={boardList}>전체조회</button>
              &nbsp;
              <button  className="btn btn-success" onClick={()=> navigate('/reboard/write')}>글쓰기</button>
            </div>            
          </div>
        </div>
      </div>        
      <Footer />
    </>
  )
}

export default ReBoardList
